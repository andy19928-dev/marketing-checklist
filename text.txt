    // ---------- 載入活動（從雲端） ----------
    async function loadActivities() {
        const { data, error } = await sb
            .from('activities')
            .select('*')
            .order('created_at', { ascending: false });

        if (error) {
            console.error(error);
            alert('讀取活動失敗，請確認 Supabase 設定是否正確。\n' + error.message);
            return;
        }
        activities = data || [];
        renderOverview();
    }

    // ---------- 渲染總覽 ----------
    function renderOverview() {
        const tbody = document.getElementById('overviewTbody');
        tbody.innerHTML = '';

        if (activities.length === 0) {
            tbody.innerHTML = `<tr><td colspan="5" style="text-align:center; color:#999;">目前沒有可顯示的活動。</td></tr>`;
            return;
        }

        activities.forEach(act => {
            const badge = act.is_public
                ? '<span class="badge badge-public">🌐 公開</span>'
                : '<span class="badge badge-private">🔒 私人</span>';

            // 上線時間僅顯示日期
            const displayLaunchTime = act.launch_time ? act.launch_time.split('T')[0] : '未設定';

            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td><strong>${act.name}</strong></td>
                <td>${act.owner || ''}</td>
                <td>${displayLaunchTime}</td>
                <td>${badge}</td>
                <td>
                    <button class="btn btn-action" onclick="viewMindmap('${act.id}')">進入心智圖檢核</button>
                    <button class="btn btn-action" style="background:#e74c3c" onclick="deleteActivity('${act.id}')">刪除</button>
                </td>`;
            tbody.appendChild(tr);
        });
    }

    // ---------- 建立新活動 ----------
    async function createActivity() {
        const name = document.getElementById('formActiveName').value.trim();
        const email = document.getElementById('formEmail').value.trim();
        
        // 處理負責人取值
        const ownerSelect = document.getElementById('formOwnerSelect').value;
        const ownerCustom = document.getElementById('formOwnerCustom').value.trim();
        const owner = ownerSelect === '自行填寫' ? ownerCustom : ownerSelect;

        // 處理簽呈字號取值
        const signNoSelect = document.getElementById('formSignNoSelect').value;
        const signNoCustom = document.getElementById('formSignNoCustom').value.trim();
        const signNo = signNoSelect === '自行填寫' ? signNoCustom : signNoSelect;

        const launchDate = document.getElementById('formLaunchTime').value;
        // 依照需求統一時間預設為 23:59:59 存入資料庫
        const launchTime = launchDate ? `${launchDate}T23:59:59` : null; 
        
        const isPublic = document.getElementById('formVisibility').value === 'true';

        if (!name || !owner || !email) {
            alert('請至少填寫活動名稱、負責人與 Email！');
            return;
        }

        const nodeStatus = {};
        for (let node in checkListTemplate) {
            nodeStatus[node] = checkListTemplate[node].map(item => ({
                text: item, checked: false, deadlineTime: ""
            }));
        }

        const payload = {
            name, owner, email,
            launch_time: launchTime,
            sign_no: signNo,
            is_public: isPublic,
            progress: nodeStatus
        };

        if (currentUser) {
            payload.owner_id = currentUser.id;
        }

        const { error } = await sb.from('activities').insert(payload);

        if (error) { alert('建立失敗：' + error.message); return; }

        closeModal('formOverlay');
        ['formActiveName', 'formOwnerCustom', 'formEmail', 'formLaunchTime', 'formSignNoCustom'].forEach(id => document.getElementById(id).value = '');
        document.getElementById('formOwnerSelect').value = '';
        document.getElementById('formSignNoSelect').value = '';
        document.getElementById('formOwnerCustom').style.display = 'none';
        document.getElementById('formSignNoCustom').style.display = 'none';
        document.getElementById('formVisibility').value = 'true';
        await loadActivities();
    }

        currentActivityId = id;

        document.getElementById('mindmapActivityTitle').innerText = `【${act.name}】流程檢核`;
        document.getElementById('mindmapActivityMeta').innerText =
            `負責人：${act.owner || '-'} (${act.email || '-'}) ｜ 簽呈字號：${act.sign_no || '無'}`;
        openModal('mindmapOverlay');
    }

    // ---------- Checklist 畫面生成邏輯 ----------
    function openChecklist(nodeName) {
        currentNodesName = nodeName;
        const act = activities.find(a => a.id === currentActivityId);

        document.getElementById('checklistOverlay').style.display = 'flex';
        document.getElementById('checklistTitle').innerText = nodeName + ' 項目檢核';

        const ul = document.getElementById('checklistContent');
        ul.innerHTML = '';
        const items = (act.progress && act.progress[nodeName]) || [];

        items.forEach((item) => {
            const li = document.createElement('li');
            li.innerHTML = generateItemHTML(item);
            ul.appendChild(li);
        });
    }

    // 動態新增一筆空白檢核項目
    function addNewChecklistItem() {
        const ul = document.getElementById('checklistContent');
        const li = document.createElement('li');
        li.innerHTML = generateItemHTML({text: '', checked: false, deadlineTime: ''});
        ul.appendChild(li);
        // Focus 到新產生的 input 上
        li.querySelector('.text-item').focus();
    }

    function generateItemHTML(item) {
        // 從資料庫取回的時間可能包含 T23:59:59，在 date 選擇器上只綁定前方的 YYYY-MM-DD
        const dateVal = item.deadlineTime ? item.deadlineTime.split('T')[0] : '';
        
        return `
            <div class="item-left">
                <input type="checkbox" class="chk-item" ${item.checked ? 'checked' : ''}>
                <input type="text" class="text-item" value="${item.text}" placeholder="點擊編輯檢核項目內容...">
            </div>
            <div style="display:flex; align-items:center;">
                <span style="font-size:11px; color:#718096; margin-right:5px; white-space:nowrap;">⏰ 截止時間:</span>
                <input type="date" class="deadline-input" value="${dateVal}">
                <span class="delete-item-btn" onclick="this.closest('li').remove()" title="刪除此項目">🗑️</span>
            </div>`;
    }

    // ---------- 儲存進度 ----------
    async function saveChecklistProgress() {
        const act = activities.find(a => a.id === currentActivityId);
        const ul = document.getElementById('checklistContent');
        
        const newItems = [];
        ul.querySelectorAll('li').forEach((li) => {
            const checked = li.querySelector('.chk-item').checked;
            const text = li.querySelector('.text-item').value.trim();
            const dateVal = li.querySelector('.deadline-input').value;
            
            // 統一預設 23:59，組成完整時間字串以便後端 24 小時前寄信判斷
            const deadlineTime = dateVal ? `${dateVal}T23:59:59` : '';

            if (text) { // 自動略過空白項目
                newItems.push({ text, checked, deadlineTime });
            }
        });

        // 寫回本地暫存
        act.progress[currentNodesName] = newItems;

        const { error } = await sb.from('activities')
            .update({ progress: act.progress })
            .eq('id', act.id);

        if (error) { alert('儲存失敗：' + error.message); return; }
        closeChecklist();
    }

    function closeChecklist() {
        document.getElementById('checklistOverlay').style.display = 'none';
    }

    // ---------- 刪除活動 ----------
    async function deleteActivity(id) {
        if (!confirm('確定要刪除這個活動及其所有檢核紀錄嗎？')) return;
        const { error } = await sb.from('activities').delete().eq('id', id);
        if (error) { alert('刪除失敗：' + error.message); return; }
        await loadActivities();
    }
</script>
